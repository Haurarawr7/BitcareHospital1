#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Edge {
    struct Edge* next_jalur;
    struct Simpul* tujuan;
    int nilai;
} Edge;

void delEdge(char tujuan, Simpul *awal){
    Edge *hapus = awal->jalur;
    if (hapus == NULL){
        printf("Tidak ada jalur simpul tujuan\n");
        return;
    }

    Edge *prev = NULL;
    int ketemu  0;

    while (hapus != NULL && ketemu == 0){
        if (hapus->tujuan->label == tujuan){
            ketemu = 1;
        }
        else {
            prev = hapus;
            hapus = hapus->next;
        }
    }

    if (ketemu) {
        if(prev == NULL){
            awal->jalur = hapus->next;
            hapus->next = NULL;
        }
        else {
            prev->next = hapus->next;
            hapus->next = NULL;
        }
    }
    free(hapus);
}
